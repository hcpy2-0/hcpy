#include <tunables/global>

profile hcpy flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Network access for MQTT + WebSocket
  network inet stream,
  network inet dgram,
  network inet6 stream,
  network inet6 dgram,
  network tcp,
  network udp,

  # Python runtime
  /usr/bin/python3* rix,
  /usr/lib/python3/** r,
  /usr/lib/python3/**/*.so mr,

  # Application files
  /app/** r,
  /app/**/*.py rix,
  /app/**/*.sh rix,
  /app/run.sh rix,

  # Configuration (addon_config)
  /config/ r,
  /config/** rw,

  # Persistent data
  /data/ rw,
  /data/** rw,

  # Temp files
  /tmp/ rw,
  /tmp/** rw,

  # Shell and utilities
  /bin/bash rix,
  /usr/bin/bashio rix,
  /bin/sh rix,
  /bin/busybox rix,
  /usr/bin/curl rix,
  /usr/bin/timeout rix,
  /usr/bin/jq rix,

  # SSL/TLS certificates
  /etc/ssl/** r,
  /usr/share/ca-certificates/** r,

  # Proc (minimal)
  /proc/*/fd/ r,
  /proc/*/mounts r,
  @{PROC}/sys/net/** r,

  # Deny sensitive paths
  deny /sys/** w,
  deny /proc/*/mem rw,
  deny /dev/** rw,
  deny /boot/** rw,
  deny /lib/modules/** rw,
}
